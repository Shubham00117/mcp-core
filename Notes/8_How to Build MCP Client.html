Math MCP Server (arith_server.py)

mcp-math-server/main.pyPython
# arith_server.py
from __future__ import annotations
from fastmcp import FastMCP

mcp = FastMCP("arith")

def _as_number(x):
    # Accept ints/floats or numeric strings; raise clean errors otherwise
    if isinstance(x, (int, float)):
        return float(x)
    if isinstance(x, str):
        return float(x.strip())
    raise TypeError("Expected a number (int/float or numeric string)")

@mcp.tool()
async def add(a: float, b: float) -> float:
    """Return a + b."""
    return _as_number(a) + _as_number(b)

@mcp.tool()
async def subtract(a: float, b: float) -> float:
    """Return a - b."""
    return _as_number(a) - _as_number(b)

@mcp.tool()
async def multiply(a: float, b: float) -> float:
    """Return a * b."""
    return _as_number(a) * _as_number(b)
💡

Note: _as_number() is a helper that normalises inputs — accepts int, float, or numeric string. This makes the server more robust when called by an LLM.

import asyncio
from langchain_mcp_adapters.client import MultiServerMCPClient

SERVERS = {
    "math": {
        "transport": "stdio",
        "command": "/Library/Frameworks/Python.framework/Versions/3.11/bin/uv",
        "args": [
            "run",
            "fastmcp",
            "run",
            "/Users/nitish/Desktop/mcp-math-server/main.py"
        ]
    }
}

async def main():
    client = MultiServerMCPClient(SERVERS)
    tools = await client.get_tools()   # returns list of LangChain StructuredTools

    # Build a name → tool lookup dict
    named_tools = {}
    for tool in tools:
        named_tools[tool.name] = tool

    print(named_tools)

if __name__ == '__main__':
    asyncio.run(main())

Full Tool Call Loop — LLM → Tool → Result → Final Answer

The 4-Step Agentic Loop
1. Prompt
→ LLM
→
2. LLM picks
tool + args
→
3. Execute
tool
→
4. LLM makes
final response
tool result wrapped in ToolMessage → passed back to LLM for final answer
client1.py — Complete Tool Call LoopPython
from langchain_openai import ChatOpenAI
from langchain_core.messages import ToolMessage
from dotenv import load_dotenv

load_dotenv()

# ... (SERVERS dict and MultiServerMCPClient setup from above)

async def main():
    client      = MultiServerMCPClient(SERVERS)
    tools       = await client.get_tools()
    named_tools = {t.name: t for t in tools}

    # Step 1 — bind tools to LLM
    llm            = ChatOpenAI(model="gpt-5")
    llm_with_tools = llm.bind_tools(tools)

    # Step 2 — send prompt, get tool selection
    prompt   = "What is the product of 12 and 15 using math tool?"
    response = await llm_with_tools.ainvoke(prompt)

    # Guard: if LLM answered directly (no tool call needed)
    if not getattr(response, "tool_calls", None):
        print("\nLLM Reply:", response.content)
        return

    # Step 3 — extract tool call details
    selected_tool    = response.tool_calls[0]["name"]
    selected_tool_args = response.tool_calls[0]["args"]
    selected_tool_id   = response.tool_calls[0]["id"]

    print(f"Selected tool: {selected_tool}")
    print(f"Selected tool args: {selected_tool_args}")

    # Step 4 — invoke the actual MCP tool
    tool_result = await named_tools[selected_tool].ainvoke(selected_tool_args)
    print(f"Tool result: {tool_result}")

    # Step 5 — wrap result and send back to LLM for final answer
    tool_message   = ToolMessage(content=tool_result, tool_call_id=selected_tool_id)
    final_response = await llm_with_tools.ainvoke([prompt, response, tool_message])
    print(f"Final response: {final_response.content}")
Selected tool: multiply
Selected tool args: {'a': 12, 'b': 15}
Tool result: 180.0
Final response: The product of 12 and 15 is 180.

response.tool_calls[0] fields

  • "name" — tool the LLM chose
  • "args" — dict of arguments
  • "id" — unique call ID (needed for ToolMessage)

ToolMessage

  • Wraps the tool execution result
  • Must include tool_call_id (from response)
  • Pass as part of conversation history: [prompt, response, tool_message]
⚠️

Always guard for no tool_calls: if the LLM can answer without a tool (e.g. "What is the capital of India"), response.tool_calls will be empty. Check with getattr(response, "tool_calls", None) before accessing index 0.

Multi-Server Support — Adding Remote Expense Server

client1.py — SERVERS with local + remotePython
load_dotenv()

SERVERS = {
    "math": {                              # Local server via STDIO
        "transport": "stdio",
        "command": "/Library/.../bin/uv",
        "args": [
            "run", "fastmcp", "run",
            "/Users/nitish/Desktop/mcp-math-server/main.py"
        ]
    },
    "expense": {                           # Remote server via HTTP
        "transport": "streamable_http",    # if this fails, try "sse"
        "url": "https://splendid-gold-dingo.fastmcp.app/mcp"
    }
}

Local Server Config

  • transport: "stdio"
  • command — path to Python/uv binary
  • args — command to run the server script
  • Server process spawned automatically

Remote Server Config

  • transport: "streamable_http"
  • url — FastMCP Cloud /mcp endpoint
  • Fallback: try "sse" if streamable_http fails
  • No local process needed
💡

How the loop handles multiple tools: get_tools() returns all tools from all servers combined. The LLM selects the right tool by name. named_tools[selected_tool] routes to the correct server automatically.

Streamlit Chat UI — MCP Chat App

Final App — localhost:8501

A Streamlit chat interface powered by the MCP client. Users type messages → the client runs the tool call loop → response displayed in chat.


Demo query: "show me all the expense from last 2 weeks"
Response: Here are your expenses from the last 2 weeks (Oct 30–Nov 12, 2025):
• Nov 04, 2025: $800.00 — Groceries

app.py — Streamlit Chat (pattern)Python
import streamlit as st
import asyncio
from langchain_mcp_adapters.client import MultiServerMCPClient
from langchain_openai import ChatOpenAI
from langchain_core.messages import ToolMessage

st.title("🧰 MCP Chat")

async def chat(user_input: str) -> str:
    client         = MultiServerMCPClient(SERVERS)
    tools          = await client.get_tools()
    named_tools    = {t.name: t for t in tools}
    llm_with_tools = ChatOpenAI(model="gpt-5").bind_tools(tools)

    response = await llm_with_tools.ainvoke(user_input)

    if not getattr(response, "tool_calls", None):
        return response.content

    tc           = response.tool_calls[0]
    tool_result  = await named_tools[tc["name"]].ainvoke(tc["args"])
    tool_msg     = ToolMessage(content=tool_result, tool_call_id=tc["id"])
    final        = await llm_with_tools.ainvoke([user_input, response, tool_msg])
    return final.content

# Chat UI loop
if prompt := st.chat_input("Type a message..."):
    reply = asyncio.run(chat(prompt))
    st.chat_message("assistant").write(reply)
🚀

Run with: streamlit run app.py — opens at localhost:8501
